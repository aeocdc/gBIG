create database componenttib;

grant all on componenttib.* to 'webservice'@'localhost' identified by 'webservice@web';

use componenttib;

#user_submit
create table user_submit(id int primary key auto_increment not null,assembleComponentList text,len_ha int,submit_time DATETIME);


#promoter
create table promoter(id int primary key auto_increment not null,name varchar(200),organism varchar(200),sequence text,parts_type varchar(200),inducible_or_constitutive varchar(200),strength_our_lab int,reference_paper text);
load data local infile 'promoter.txt' into table promoter fields terminated by '\t' ignore 1 lines (organism,parts_type,name,sequence,inducible_or_constitutive,strength_our_lab,reference_paper);

#promoter_function_module
create table promoter_function_module(id int primary key auto_increment not null,name varchar(200),organism varchar(200),sequence text,parts_type varchar(200),inducible_or_constitutive varchar(200),strength_our_lab int,reference_paper text,original_source varchar(200));
